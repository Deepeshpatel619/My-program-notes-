STL-list class

STL list template class used for creating lists where insertion or deletion can take place from both ends (front,end) but elements are accessed sequentially only

                
#include < iostream>
#include < list>
using namespace std;

int main() {
 
list < int> l={1,6,3};
    
l.push_back(10);// add element at last 
l.push_front(5);//add element at front 
l.pop_back();//remove last element
l.pop_front();//remove first element 
l.sort();//sorts Elements 
cout<< l.size()<< endl;//return 
size of list
l.reverse();//reverse list elements 
l.remove(3);//remove specific
 element from list 
// l.clear(); removesall element
of list 
    
list< int>::iterator p=l.begin();
while(p!=l.end())
{
   cout<< *p<< ",";
   p++;
 }
    
    
  return 0;
}
